\chapter{Marketing}

\section{Angebot}

\subsection{Kundennutzen}
\begin{itemize}
    \item Überblick über alle Lebensmittelvorräte im Haushalt
    \item Rechtzeitige Erinnerung an ablaufende Produkte durch farbliche Warnungen
    \item Weniger Lebensmittelverschwendung = Geldersparnis
    \item Gemeinsame Verwaltung für alle Haushaltsmitglieder
\end{itemize}

\subsection{Vorteile gegenüber der Konkurrenz}
\begin{itemize}
    \item Echtes Multi-User-System mit Rollen (Eigentümer, Co-Eigentümer, Mitwirkende)
    \item DSGVO-konform mit deutschem Serverstandort
    \item Keine Installation nötig -- funktioniert im Browser
\end{itemize}

\subsection{Serviceleistungen}
\begin{itemize}
    \item Online-Hilfe und FAQ-Bereich
    \item E-Mail-Support für alle Nutzer
    \item Priority-Support für Pro-Kunden
    \item Regelmäßige Updates und neue Features
\end{itemize}

\subsection{Garantieleistungen}
\begin{itemize}
    \item 30 Tage Geld-zurück-Garantie bei Premium-Abos
    \item Datensicherung und Verfügbarkeitsgarantie (99\% Uptime)
    \item Kostenloser Datenexport jederzeit möglich
\end{itemize}
